    Agent¶

    class Agent¶

    核心 Agent 类，实现 AI 代理的完整功能。

    from agentica import Agent

    构造参数¶

    参数 类型 默认值 描述
    model Model None LLM 模型实例
    name str None Agent 名称
    agent_id str None Agent 唯一标识（自动生成 UUID）
    user_id str None 用户 ID
    session_id str None 会话 ID
    description str None Agent 描述
    instructions str \| List[str] \| Callable None 指令/提示词
    system_prompt str \| Callable None 系统提示词
    tools List[Tool \| Callable] None 工具列表
    knowledge Knowledge None 知识库
    memory AgentMemory AgentMemory() 内存管理器
    db BaseDb None 持久化数据库
    team List[Agent] None Agent 团队
    role str None 团队中的角色
    enable_multi_round bool False 启用多轮对话策略
    max_rounds int 100 最大对话轮数
    max_tokens int 128000 最大 token 数
    response_model Type[BaseModel] None 结构化输出模型
    structured_outputs bool False 使用结构化输出
    add_history_to_messages bool False 添加历史到消息
    num_history_responses int 3 历史响应数量
    add_references bool False 添加知识库引用
    markdown bool False Markdown 格式输出
    debug bool False 调试模式

    方法¶

    async run(message, **kwargs) -> RunResponse¶

    异步运行 Agent（非流式）。

    # 异步非流式
response = await agent.run("你好")
print(response.content)

    参数: - message: str | List | Dict | Message - 输入消息 - **kwargs: 其他参数传递给模型

    返回: RunResponse

    async run_stream(message, **kwargs) -> AsyncIterator[RunResponse]¶

    异步流式运行 Agent。

    # 异步流式
async for chunk in agent.run_stream("你好"):
    print(chunk.content, end="")

    参数: - message: str | List | Dict | Message - 输入消息 - **kwargs: 其他参数传递给模型

    返回: AsyncIterator[RunResponse]

    run_sync(message, **kwargs) -> RunResponse¶

    同步运行 Agent（非流式）。内部调用 run() 的同步适配器。

    # 同步非流式
response = agent.run_sync("你好")
print(response.content)

    参数: - message: str | List | Dict | Message - 输入消息 - **kwargs: 其他参数传递给模型

    返回: RunResponse

    run_stream_sync(message, **kwargs) -> Iterator[RunResponse]¶

    同步流式运行 Agent。内部通过后台线程驱动异步迭代器。

    # 同步流式
for chunk in agent.run_stream_sync("你好"):
    print(chunk.content, end="")

    参数: - message: str | List | Dict | Message - 输入消息 - **kwargs: 其他参数传递给模型

    返回: Iterator[RunResponse]

    async print_response(message, **kwargs) / async print_response_stream(message, **kwargs)¶

    异步打印格式化的响应（非流式 / 流式）。

    # 非流式
await agent.print_response("解释量子计算")

# 流式
await agent.print_response_stream("解释量子计算")
    print_response_sync(message, **kwargs) / print_response_stream_sync(message, **kwargs)¶

    同步打印格式化的响应（非流式 / 流式）。

    # 非流式
agent.print_response_sync("解释量子计算")

# 流式
agent.print_response_stream_sync("解释量子计算")

    class Tool¶

    创建自定义工具¶

    方式 1: 使用函数

    def get_weather(city: str) -> str:
    """获取城市天气

    Args:
        city: 城市名称
    """
    return f"{city}天气晴朗，25°C"

agent = Agent(tools=[get_weather])

    方式 2: 继承 Tool 类

    from agentica import Tool

class WeatherTool(Tool):
    def __init__(self):
        super().__init__(name="weather")
        self.register(self.get_weather)

    def get_weather(self, city: str) -> str:
        """获取城市天气"""
        return f"{city}天气晴朗"

    Workflow¶

    class Workflow¶

    工作流引擎。

    from agentica import Workflow

    创建工作流¶

    from agentica import Workflow, Agent, RunResponse
from pydantic import Field

class MyWorkflow(Workflow):
    researcher: Agent = Field(...)
    writer: Agent = Field(...)

    async def run(self, topic: str) -> RunResponse:
        # 研究阶段
        research = await self.researcher.run(f"研究: {topic}")

        # 写作阶段
        article = await self.writer.run(f"基于以下研究写文章:\n{research.content}")

        return RunResponse(content=article.content)

# 使用
workflow = MyWorkflow(
    researcher=Agent(name="Researcher", ...),
    writer=Agent(name="Writer", ...),
)

# 异步运行
result = await workflow.run("人工智能的未来")

# 同步运行（使用适配器）
result = workflow.run_sync("人工智能的未来")

    class MCPConfig¶

    MCP 服务器配置。

config = MCPConfig(
    servers=[
        {
            "name": "filesystem",

    ]
)

agent = Agent(mcp_config=config)
